Functions & Methods

Create reusable code blocks with methods. Understand parameters, return values, and scope.

1. Method Definition and Calling

Methods are reusable blocks of code that perform specific tasks. In Java, all functions are methods of a class.

Example 1: Basic Method Definition

public class MethodDemo {
    // Define a simple method
    public static void greet() {
        System.out.println("Hello, World!");
    }
    
    public static void main(String[] args) {
        // Call the method
        greet();  // Output: Hello, World!
    }
}

// Output: Hello, World!

Example 2: Method with Multiple Statements

public class Calculator {
    public static void displayInfo() {
        System.out.println("Welcome to Java methods!");
        System.out.println("Methods help organize code.");
        System.out.println("They make code reusable.");
    }
    
    public static void calculateSquare() {
        int number = 5;
        int square = number * number;
        System.out.println("Square of " + number + " is " + square);
    }
    
    public static void main(String[] args) {
        displayInfo();
        calculateSquare();  // Output: Square of 5 is 25
    }
}

Example 3: Method Naming Conventions

public class BestPractices {
    // Method names should be descriptive and in camelCase
    public static void calculateArea() {
        int length = 10;
        int width = 5;
        int area = length * width;
        System.out.println("Area: " + area);
    }
    
    // Method can be called multiple times
    public static void sayHello() {
        System.out.println("Hello!");
    }
    
    // Methods can call other methods
    public static void morningRoutine() {
        sayHello();
        System.out.println("Time for breakfast!");
    }
    
    public static void main(String[] args) {
        calculateArea();      // Area: 50
        morningRoutine();
    }
}

2. Parameters and Arguments

Parameters are variables in method definition. Arguments are values passed when calling the method.

Example 1: Method with Parameters

public class ParameterDemo {
    public static void greetPerson(String name, int age) {
        System.out.println("Hello, " + name + "! You are " + age + " years old.");
    }
    
    public static void calculateRectangleArea(int length, int width) {
        int area = length * width;
        System.out.println("Area: " + area);
    }
    
    public static void main(String[] args) {
        greetPerson("Alice", 25);  
        // Output: Hello, Alice! You are 25 years old.
        
        greetPerson("Bob", 30);
        // Output: Hello, Bob! You are 30 years old.
        
        calculateRectangleArea(10, 5);  // Area: 50
    }
}

Example 2: Order of Parameters Matters

public class DivisionDemo {
    public static void divide(double a, double b) {
        double result = a / b;
        System.out.println(a + " divided by " + b + " is " + result);
    }
    
    public static void main(String[] args) {
        divide(10, 2);   // 10 divided by 2 is 5.0
        divide(2, 10);   // 2 divided by 10 is 0.2
    }
}

Example 3: Variable Number of Arguments (Varargs)

public class VarargsDemo {
    // ... int (varargs - variable arguments)
    public static int sumAll(int... numbers) {
        int total = 0;
        for (int num : numbers) {
            total += num;
        }
        return total;
    }
    
    public static void printItems(String... items) {
        for (String item : items) {
            System.out.println("- " + item);
        }
    }
    
    public static void main(String[] args) {
        System.out.println(sumAll(1, 2, 3));           // 6
        System.out.println(sumAll(10, 20, 30, 40));   // 100
        System.out.println(sumAll());                  // 0
        
        printItems("Apple", "Banana", "Orange");
    }
}

3. Return Statements

Methods can return values using the return statement. Return ends method execution.

Example 1: Returning Values

public class ReturnDemo {
    public static int addNumbers(int a, int b) {
        return a + b;
    }
    
    public static String getUserInfo() {
        return "Alice, 25, Engineer";
    }
    
    public static double calculateAverage(double a, double b, double c) {
        return (a + b + c) / 3;
    }
    
    public static void main(String[] args) {
        int result = addNumbers(5, 3);
        System.out.println("Sum: " + result);  // Sum: 8
        
        String info = getUserInfo();
        System.out.println(info);
        
        double avg = calculateAverage(10, 20, 30);
        System.out.println("Average: " + avg);  // Average: 20.0
    }
}

Example 2: Multiple Return Points

public class ConditionalReturn {
    public static String checkNumber(int num) {
        if (num > 0) {
            return "Positive";
        } else if (num < 0) {
            return "Negative";
        } else {
            return "Zero";
        }
    }
    
    public static String validateAge(int age) {
        if (age < 0) {
            return "Invalid age";
        }
        if (age < 18) {
            return "Minor";
        }
        return "Adult";
    }
    
    public static void main(String[] args) {
        System.out.println(checkNumber(5));    // Positive
        System.out.println(checkNumber(-3));   // Negative
        System.out.println(checkNumber(0));    // Zero
        
        System.out.println(validateAge(25));   // Adult
        System.out.println(validateAge(15));   // Minor
    }
}

Example 3: Methods that Return Nothing (void)

public class VoidMethodDemo {
    // void means method doesn't return a value
    public static void printSquare(int num) {
        System.out.println(num * num);
    }
    
    // Returns a value
    public static int calculateSquare(int num) {
        return num * num;
    }
    
    // Practical calculator function
    public static double calculator(double a, double b, String operation) {
        switch(operation) {
            case "add":
                return a + b;
            case "subtract":
                return a - b;
            case "multiply":
                return a * b;
            case "divide":
                if (b != 0) {
                    return a / b;
                } else {
                    return Double.NaN;  // Not a Number
                }
            default:
                return 0;
        }
    }
    
    public static void main(String[] args) {
        printSquare(4);  // Output: 16
        
        int result = calculateSquare(4);
        System.out.println("Square: " + result);  // Square: 16
        
        System.out.println(calculator(10, 5, "add"));      // 15.0
        System.out.println(calculator(10, 5, "divide"));   // 2.0
    }
}

4. Local and Static Variables

Local variables exist within methods. Static variables belong to the class and are shared across all instances.

Example 1: Local Variables

public class LocalVariableDemo {
    public static void myMethod() {
        int localVar = 10;  // Local variable
        System.out.println("Local: " + localVar);
    }
    
    // This would cause an error:
    // System.out.println(localVar);  // Error: localVar not defined
    
    public static void function1() {
        int x = 10;
        System.out.println("function1 x: " + x);
    }
    
    public static void function2() {
        int x = 20;  // Different x in function2
        System.out.println("function2 x: " + x);
    }
    
    public static void main(String[] args) {
        myMethod();      // Local: 10
        function1();     // function1 x: 10
        function2();     // function2 x: 20
    }
}

Example 2: Static Variables

public class StaticVariableDemo {
    // Static variable - shared across all instances
    static int count = 0;
    
    public static void increment() {
        count++;
        System.out.println("Count: " + count);
    }
    
    public static void main(String[] args) {
        increment();  // Count: 1
        increment();  // Count: 2
        increment();  // Count: 3
        
        System.out.println("Final count: " + count);  // Final count: 3
    }
}

Example 3: Instance Variables vs Static Variables

public class Counter {
    int instanceCount = 0;  // Each object has its own
    static int staticCount = 0;   // Shared by all objects
    
    public void incrementBoth() {
        instanceCount++;
        staticCount++;
    }
    
    public void display() {
        System.out.println("Instance: " + instanceCount + ", Static: " + staticCount);
    }
    
    public static void main(String[] args) {
        Counter c1 = new Counter();
        Counter c2 = new Counter();
        
        c1.incrementBoth();
        System.out.println("After c1.increment():");
        c1.display();  // Instance: 1, Static: 1
        c2.display();  // Instance: 0, Static: 1
        
        c2.incrementBoth();
        System.out.println("After c2.increment():");
        c1.display();  // Instance: 1, Static: 2
        c2.display();  // Instance: 1, Static: 2
    }
}
