Control Flow

Control program execution with if-elif-else statements and understand conditional logic.

1. If-Else Statements

If-else statements allow your program to make decisions and execute different code based on conditions.

Example 1: Basic If Statement

# Simple if statement
age = 20

if age >= 18:
    print("You are an adult")
    print("You can vote")

# Output:
# You are an adult
# You can vote

# If condition is False, code block is skipped
age = 15

if age >= 18:
    print("You are an adult")  # This won't execute

print("Program continues...")  # This executes

Example 2: If-Else Statement

# If-else provides alternative execution path
temperature = 75

if temperature > 80:
    print("It's hot outside!")
else:
    print("The weather is pleasant")

# Output: The weather is pleasant

# Account balance check
balance = 50
withdrawal = 75

if withdrawal <= balance:
    balance = balance - withdrawal
    print(f"Withdrawal successful. New balance: ${balance}")
else:
    print("Insufficient funds!")
    print(f"Your balance: ${balance}")

# Output: Insufficient funds!
#         Your balance: $50

Example 3: Multiple Conditions with Logical Operators

# Using 'and' operator
age = 25
has_license = True

if age >= 18 and has_license:
    print("You can drive a car")
else:
    print("You cannot drive")

# Output: You can drive a car

# Using 'or' operator
day = "Saturday"
is_holiday = False

if day == "Saturday" or day == "Sunday" or is_holiday:
    print("It's a day off!")
else:
    print("It's a working day")

# Output: It's a day off!

# Complex condition
score = 85
attendance = 90

if score >= 80 and attendance >= 85:
    print("Congratulations! You qualify for honors")
else:
    print("Keep working hard!")

# Output: Congratulations! You qualify for honors

Example 4: Password Validator

# Real-world example: Password validation
password = "SecurePass123"

# Check multiple conditions
has_length = len(password) >= 8
has_digit = any(char.isdigit() for char in password)
has_upper = any(char.isupper() for char in password)

if has_length and has_digit and has_upper:
    print("Strong password!")
    print("Password accepted")
else:
    print("Weak password!")
    if not has_length:
        print("- Password must be at least 8 characters")
    if not has_digit:
        print("- Password must contain at least one digit")
    if not has_upper:
        print("- Password must contain at least one uppercase letter")

# Output: Strong password!
#         Password accepted

2. Elif Chains for Multiple Conditions

Use elif (else if) to check multiple conditions in sequence.

Example 1: Grade Calculator

# Determine letter grade based on score
score = 87

if score >= 90:
    grade = 'A'
    print(f"Excellent! Grade: {grade}")
elif score >= 80:
    grade = 'B'
    print(f"Good job! Grade: {grade}")
elif score >= 70:
    grade = 'C'
    print(f"Fair. Grade: {grade}")
elif score >= 60:
    grade = 'D'
    print(f"Needs improvement. Grade: {grade}")
else:
    grade = 'F'
    print(f"Failed. Grade: {grade}")

# Output: Good job! Grade: B

# Only ONE block executes (the first True condition)
# Remaining conditions are not checked

Example 2: Ticket Pricing System

# Calculate ticket price based on age
age = 8

if age < 3:
    price = 0
    category = "Free (Infant)"
elif age < 12:
    price = 10
    category = "Child"
elif age < 18:
    price = 15
    category = "Teen"
elif age < 65:
    price = 20
    category = "Adult"
else:
    price = 12
    category = "Senior"

print(f"Age: {age}")
print(f"Category: {category}")
print(f"Ticket Price: ${price}")

# Output:
# Age: 8
# Category: Child
# Ticket Price: $10

Example 3: BMI Calculator with Categories

3. Nested Conditionals

Place if statements inside other if statements for complex decision-making.

Example 1: Basic Nested If

# Check multiple levels of conditions
age = 25
has_ticket = True

if age >= 18:
    print("You meet the age requirement")
    
    if has_ticket:
        print("You can enter the concert")
        print("Enjoy the show!")
    else:
        print("Please purchase a ticket first")
else:
    print("Sorry, you must be 18 or older")

# Output:
# You meet the age requirement
# You can enter the concert
# Enjoy the show!

Example 2: Login System

# Multi-level authentication check
username = "admin"
password = "secret123"
is_verified = True

if username == "admin":
    print("Username correct")
    
    if password == "secret123":
        print("Password correct")
        
        if is_verified:
            print("Login successful!")
            print("Welcome to the admin panel")
        else:
            print("Please verify your email first")
    else:
        print("Incorrect password")
        print("Please try again")
else:
    print("Username not found")

# Output:
# Username correct
# Password correct
# Login successful!
# Welcome to the admin panel

Example 3: Loan Approval System

# Complex nested conditions for loan approval
age = 30
income = 55000
credit_score = 720
employment_years = 5

print("Loan Application Review")
print("-" * 30)

if age >= 21 and age <= 65:
    print("✓ Age requirement met")
    
    if income >= 30000:
        print("✓ Income requirement met")
        
        if credit_score >= 650:
            print("✓ Credit score acceptable")
            
            if employment_years >= 2:
                print("✓ Employment history sufficient")
                print("\n✅ LOAN APPROVED!")
                
                # Determine interest rate
                if credit_score >= 750:
                    rate = 3.5
                elif credit_score >= 700:
                    rate = 4.5

                    rate = 5.5
                
                print(f"Interest Rate: {rate}%")
            else:
                print("✗ Need at least 2 years employment")

        else:
            print("✗ Credit score too low")
            print("\n❌ LOAN DENIED")
    else:
        print("✗ Insufficient income")
        print("\n❌ LOAN DENIED")
else:
    print("✗ Age must be between 21 and 65")
    print("\n❌ LOAN DENIED")

Example 4: Shipping Cost Calculator

# Calculate shipping based on weight and destination
weight = 5.5  # kg
destination = "International"
is_express = True

print(f"Package Weight: {weight}kg")
print(f"Destination: {destination}")

if destination == "Domestic":
    if weight <= 1:
        base_cost = 5
    elif weight <= 5:
        base_cost = 10
    else:
        base_cost = 15
    
    if is_express:
        total_cost = base_cost * 1.5
        delivery = "1-2 days"
    else:
        total_cost = base_cost
        delivery = "3-5 days"

elif destination == "International":
    if weight <= 1:
        base_cost = 20
    elif weight <= 5:
        base_cost = 40
    else:
        base_cost = 60
    
    if is_express:
        total_cost = base_cost * 1.8
        delivery = "3-5 days"
    else:
        total_cost = base_cost
        delivery = "7-14 days"
else:
    total_cost = 0
    delivery = "Unknown"

print(f"Shipping Cost: ${total_cost:.2f}")
print(f"Estimated Delivery: {delivery}")

# Output:
# Package Weight: 5.5kg
# Destination: International
# Shipping Cost: $108.00
# Estimated Delivery: 3-5 days

4. Ternary Operators

Write compact conditional expressions in a single line.

Example 1: Basic Ternary Operator

# Syntax: value_if_true if condition else value_if_false

# Traditional if-else
age = 20
if age >= 18:
    status = "Adult"
else:
    status = "Minor"
print(status)  # Output: Adult

# Using ternary operator (more concise)
age = 20
status = "Adult" if age >= 18 else "Minor"
print(status)  # Output: Adult

# Another example
score = 85
result = "Pass" if score >= 50 else "Fail"
print(f"Result: {result}")  # Output: Result: Pass

# With numbers
a = 10
b = 20
max_value = a if a > b else b
print(f"Maximum: {max_value}")  # Output: Maximum: 20

Example 2: Ternary in Print Statements

# Use ternary directly in print
temperature = 75
print(f"It's {'hot' if temperature > 80 else 'pleasant'}")
# Output: It's pleasant

# Even/Odd checker
number = 7
print(f"{number} is {'even' if number % 2 == 0 else 'odd'}")
# Output: 7 is odd

# Price with discount
price = 100
has_coupon = True
final_price = price * 0.9 if has_coupon else price
print(f"Final price: ${final_price}")
# Output: Final price: $90.0

# Greeting based on time
hour = 14
greeting = "Good morning" if hour < 12 else "Good afternoon" if hour < 18 else "Good evening"
print(greeting)  # Output: Good afternoon

Example 3: Nested Ternary Operators

# Multiple conditions in ternary (use carefully!)
score = 85

# Traditional nested if
if score >= 90:
    grade = 'A'
elif score >= 80:
    grade = 'B'
elif score >= 70:
    grade = 'C'
else:
    grade = 'F'

# Using nested ternary (less readable but compact)
grade = 'A' if score >= 90 else 'B' if score >= 80 else 'C' if score >= 70 else 'F'
print(f"Grade: {grade}")  # Output: Grade: B

# Tip calculator
bill = 50
service = "excellent"  # excellent, good, poor

# Nested ternary for tip percentage
tip_percent = 0.20 if service == "excellent" else 0.15 if service == "good" else 0.10
tip = bill * tip_percent
print(f"Tip: ${tip:.2f}")  # Output: Tip: $10.00

Example 4: Practical Ternary Examples

# Absolute value
number = -15
absolute = number if number >= 0 else -number
print(f"Absolute value: {absolute}")  # Output: 15

# Pluralization
count = 1
print(f"{count} item{'s' if count != 1 else ''}")
# Output: 1 item

count = 5
print(f"{count} item{'s' if count != 1 else ''}")
# Output: 5 items

# User status
is_online = True
status_icon = "🟢" if is_online else "🔴"
status_text = "Online" if is_online else "Offline"
print(f"{status_icon} {status_text}")
# Output: 🟢 Online

# Discount eligibility
purchases = 3
is_member = True
discount = 0.15 if is_member and purchases > 2 else 0.10 if is_member else 0
print(f"Your discount: {discount * 100}%")
# Output: Your discount: 15.0%

# Safe division
a = 10
b = 0
result = a / b if b != 0 else "Cannot divide by zero"
print(result)  # Output: Cannot divide by zero
