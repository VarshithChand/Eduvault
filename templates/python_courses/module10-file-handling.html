File Handling

Read, write, append, and manage files in Python. Work with file paths and the os module.

1. Opening and Closing Files

Python uses the built-in open() function to work with files. Always close files after use — or better, use the with statement.

Example 1: open() and close()

# Basic open and close
file = open("data.txt", "r")   # open in read mode
content = file.read()
print(content)
file.close()                   # always close manually

# What happens if you forget to close?
# The file stays locked, changes may not be saved,
# and you risk running out of file handles.

# Opening a file that doesn't exist in read mode raises an error
# open("missing.txt", "r")  # ❌ FileNotFoundError

# Check before opening
import os
if os.path.exists("data.txt"):
    file = open("data.txt", "r")
    print(file.read())
    file.close()
else:
    print("File not found")

Example 2: The with Statement (Recommended)

# 'with' automatically closes the file — even if an error occurs
with open("data.txt", "r") as file:
    content = file.read()
    print(content)
# File is closed here automatically

# Verify the file is closed
with open("data.txt", "r") as file:
    print(file.closed)   # False — still open inside block

print(file.closed)       # True — closed after block

# Multiple files at once using with
with open("input.txt", "r") as infile, open("output.txt", "w") as outfile:
    data = infile.read()
    outfile.write(data)
    print("Copied successfully")

Example 3: File Object Attributes

# File objects have useful attributes
with open("data.txt", "r") as file:
    print(file.name)      # data.txt
    print(file.mode)      # r
    print(file.closed)    # False

# After the block
print(file.closed)        # True

# You can reopen a closed file
file = open("data.txt", "r")
file.close()
print(file.closed)        # True
file = open("data.txt", "r")  # reopen
print(file.read())
file.close()

# File pointer position
with open("data.txt", "r") as file:
    print(file.tell())    # 0 — at the start
    file.read(5)          # read 5 characters
    print(file.tell())    # 5 — pointer moved
    file.seek(0)          # go back to start
    print(file.tell())    # 0

2. File Modes

The mode argument controls how a file is opened — for reading, writing, appending, or binary operations.

Example 1: Text Modes

# "r"  — Read (default). File must exist.
with open("data.txt", "r") as f:
    print(f.read())

# "w"  — Write. Creates file if not found. Overwrites existing content!
with open("output.txt", "w") as f:
    f.write("This replaces all previous content.")

# "a"  — Append. Creates file if not found. Adds to end.
with open("log.txt", "a") as f:
    f.write("New log entry\n")

# "x"  — Create. Fails if file already exists.
# open("new.txt", "x")  # ❌ FileExistsError if new.txt exists

# "r+" — Read and Write. File must exist.
with open("data.txt", "r+") as f:
    content = f.read()
    f.seek(0)
    f.write("Updated: " + content)

Example 2: Binary Modes

# "rb" — Read Binary (images, PDFs, audio, etc.)
with open("image.png", "rb") as f:
    data = f.read()
    print(f"File size: {len(data)} bytes")

# "wb" — Write Binary
with open("copy.png", "wb") as out:
    out.write(data)
print("Image copied successfully")

# "ab" — Append Binary
with open("log.bin", "ab") as f:
    f.write(b"\x00\x01\x02")

# Binary vs text — key difference
# Text mode:   reads/writes strings (str)
# Binary mode: reads/writes bytes (bytes)

# Text
with open("data.txt", "r") as f:
    content = f.read()
    print(type(content))   # 

# Binary
with open("data.txt", "rb") as f:
    content = f.read()
    print(type(content))   #

Example 3: Mode Summary and Safe Patterns

# Mode quick reference
# ┌──────┬───────────────────────────────────┐
# │ Mode │ Description                        │
# ├──────┼───────────────────────────────────┤
# │  r   │ Read text (default)                │
# │  w   │ Write text (overwrites!)           │
# │  a   │ Append text                        │
# │  x   │ Create new (error if exists)       │
# │  r+  │ Read + write                       │
# │  rb  │ Read binary                        │
# │  wb  │ Write binary                       │
# └──────┴───────────────────────────────────┘

# Safe write — only write if file doesn't already exist
import os

filename = "config.txt"
if not os.path.exists(filename):
    with open(filename, "w") as f:
        f.write("default_settings=True\n")
    print("Config created")
else:
    print("Config already exists — skipping")

# Safe read with fallback
try:
    with open("settings.txt", "r") as f:
        print(f.read())
except FileNotFoundError:
    print("Settings file missing — using defaults")

3. Reading Files

Python provides multiple methods to read file content — all at once, line by line, or into a list.

Example 1: read() and readline()

# .read() — reads entire file as one string
with open("data.txt", "r") as file:
    content = file.read()
    print(content)

# .read(n) — reads n characters
with open("data.txt", "r") as file:
    chunk = file.read(10)   # first 10 characters
    print(chunk)

# .readline() — reads one line at a time
with open("data.txt", "r") as file:
    line1 = file.readline()   # first line
    line2 = file.readline()   # second line
    print(line1)
    print(line2)

# Loop with readline
with open("data.txt", "r") as file:
    line = file.readline()
    while line:
        print(line.strip())   # .strip() removes \n
        line = file.readline()

Example 2: readlines() and Iterating

# .readlines() — returns all lines as a list
with open("data.txt", "r") as file:
    lines = file.readlines()

print(lines)          # ['line1\n', 'line2\n', 'line3\n']
print(len(lines))     # number of lines

# Process each line
for line in lines:
    print(line.strip())

# Best way — iterate directly over file object (memory efficient)
with open("data.txt", "r") as file:
    for line in file:
        print(line.strip())

# Read into a clean list (no newline characters)
with open("data.txt", "r") as file:
    clean_lines = [line.strip() for line in file]
print(clean_lines)

Example 3: Reading and Processing Data

# Practical: read a CSV-like file and parse it
# Assume scores.txt contains:
# Alice,95
# Bob,83
# Carol,91

with open("scores.txt", "r") as file:
    students = {}
    for line in file:
        line = line.strip()
        if line:                         # skip empty lines
            name, score = line.split(",")
            students[name] = int(score)

print(students)
# {'Alice': 95, 'Bob': 83, 'Carol': 91}

# Find top student
top = max(students, key=students.get)
print(f"Top student: {top} with {students[top]}")

# Calculate average
avg = sum(students.values()) / len(students)
print(f"Class average: {avg:.1f}")

4. Writing, Appending, and File Paths

Write new content or add to existing files. Use the os and pathlib modules to manage file paths safely.

Example 1: Writing Files

# .write() — writes a string (no automatic newline)
with open("notes.txt", "w") as file:
    file.write("Line one\n")
    file.write("Line two\n")
    file.write("Line three\n")

# .writelines() — writes a list of strings
lines = ["Alice,95\n", "Bob,83\n", "Carol,91\n"]
with open("scores.txt", "w") as file:
    file.writelines(lines)

# Write with newline join
students = ["Alice", "Bob", "Carol", "Dave"]
with open("students.txt", "w") as file:
    file.write("\n".join(students))

# Overwrite warning — "w" erases existing content
with open("important.txt", "w") as f:
    f.write("First write\n")
with open("important.txt", "w") as f:
    f.write("Second write\n")   # First write is gone!

Example 2: Appending to Files

# "a" mode adds to end without deleting existing content
with open("log.txt", "a") as file:
    file.write("User logged in\n")

# Append multiple entries
import datetime

def log_event(event):
    timestamp = datetime.datetime.now().strftime("%Y-%m-%d %H:%M:%S")
    with open("log.txt", "a") as file:
        file.write(f"[{timestamp}] {event}\n")

log_event("App started")
log_event("User Alice logged in")
log_event("File uploaded")

# Read back the log
with open("log.txt", "r") as file:
    print(file.read())

Example 3: Working with File Paths (os and pathlib)

import os
from pathlib import Path

# Check if file or directory exists
print(os.path.exists("data.txt"))       # True or False
print(os.path.isfile("data.txt"))       # True if it's a file
print(os.path.isdir("my_folder"))       # True if it's a folder

# Get file info
print(os.path.getsize("data.txt"))      # size in bytes
print(os.path.abspath("data.txt"))      # absolute path

# Create and remove directories
os.makedirs("output/reports", exist_ok=True)  # create nested dirs
os.remove("old_file.txt")               # delete a file

# List files in a directory
for filename in os.listdir("."):
    if filename.endswith(".txt"):
        print(filename)

# Modern approach — pathlib (Python 3.4+)
path = Path("data.txt")
print(path.exists())          # True/False
print(path.suffix)            # .txt
print(path.stem)              # data
print(path.parent)            # current directory

# Safe read with pathlib
if path.exists():
    content = path.read_text()
    print(content)
