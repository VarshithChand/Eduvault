Sets & Advanced Collections

Work with unordered unique collections, perform powerful set operations, and explore Python's collections module.

1. Creating and Accessing Sets

A set is an unordered collection of unique elements. Duplicates are automatically removed, and sets are not indexed.

Example 1: Creating Sets

# Creating a set with curly braces
numbers = {1, 2, 3, 4, 5}
print(numbers)  # {1, 2, 3, 4, 5}

# Duplicates are automatically removed
values = {1, 2, 2, 3, 3, 3, 4}
print(values)  # {1, 2, 3, 4}

# Creating a set from a list using set()
letters = set(["a", "b", "c", "a", "b"])
print(letters)  # {'a', 'b', 'c'}

# Creating a set from a string
chars = set("hello")
print(chars)  # {'h', 'e', 'l', 'o'} — duplicates removed

# Creating an empty set — must use set(), not {}
empty_set = set()
print(type(empty_set))   # 
print(type({}))          #  — this is a dict!

# Sets can hold mixed types (all must be immutable)
mixed = {1, "hello", 3.14, True}
print(mixed)

Example 2: Checking Membership

# Use 'in' to check membership — very fast with sets
fruits = {"apple", "banana", "cherry", "mango"}

print("apple" in fruits)    # True
print("grape" in fruits)    # False
print("banana" not in fruits)  # False

# Sets are faster than lists for membership checks
# Especially useful with large collections
large_set = set(range(1000000))
print(999999 in large_set)  # True — near instant

# len() returns the number of elements
colors = {"red", "green", "blue"}
print(len(colors))  # 3

# Sets are unordered — no indexing
# colors[0]  # ❌ TypeError: 'set' object is not subscriptable

# Iterate over a set
for fruit in fruits:
    print(fruit)  # order is not guaranteed

Example 3: Sets vs Lists vs Tuples

# List — ordered, allows duplicates, mutable
my_list = [1, 2, 2, 3]
print(my_list)        # [1, 2, 2, 3]
print(my_list[0])     # 1 — indexing works

# Tuple — ordered, allows duplicates, immutable
my_tuple = (1, 2, 2, 3)
print(my_tuple)       # (1, 2, 2, 3)
print(my_tuple[0])    # 1 — indexing works

# Set — unordered, unique elements only, mutable
my_set = {1, 2, 2, 3}
print(my_set)         # {1, 2, 3}
# my_set[0]  ❌ — no indexing

# When to use a set:
# - You need unique values
# - You need fast membership testing
# - You want to perform set math (union, intersection, etc.)

# Convert between types
nums = [1, 2, 2, 3, 3, 4]
unique = list(set(nums))  # Remove duplicates from a list
print(unique)  # [1, 2, 3, 4]

2. Adding, Removing, and Modifying Sets

Sets are mutable — you can add or remove elements, but you cannot change an existing element directly.

Example 1: Adding Elements

# .add() adds a single element
fruits = {"apple", "banana"}

fruits.add("orange")
print(fruits)  # {'apple', 'banana', 'orange'}

# Adding a duplicate has no effect
fruits.add("apple")
print(fruits)  # {'apple', 'banana', 'orange'} — unchanged

# .update() adds multiple elements from any iterable
fruits.update(["mango", "grape"])
print(fruits)

# .update() can take sets, lists, or tuples
fruits.update({"kiwi"}, ("pear",))
print(fruits)

# Adding elements from another set
new_fruits = {"lemon", "lime"}
fruits.update(new_fruits)
print(len(fruits))  # total unique fruits

Example 2: Removing Elements

# .remove() — raises KeyError if not found
fruits = {"apple", "banana", "orange", "mango"}

fruits.remove("banana")
print(fruits)

# fruits.remove("grape")  # ❌ KeyError: 'grape'

# .discard() — safe, no error if element not found
fruits.discard("mango")
print(fruits)

fruits.discard("grape")  # ✅ no error
print(fruits)            # unchanged

# .pop() removes and returns an arbitrary element
fruits = {"apple", "banana", "cherry"}
removed = fruits.pop()
print(f"Removed: {removed}")
print(f"Remaining: {fruits}")

# .clear() removes all elements
data = {1, 2, 3, 4, 5}
data.clear()
print(data)  # set()

Example 3: Copying Sets

# .copy() creates a shallow copy
original = {1, 2, 3, 4}
copy = original.copy()

copy.add(5)
print(original)  # {1, 2, 3, 4} — unchanged
print(copy)      # {1, 2, 3, 4, 5}

# Assignment does NOT copy — it creates a reference
a = {10, 20, 30}
b = a           # b points to the same set
b.add(40)
print(a)        # {10, 20, 30, 40} — a is also changed!

# Always use .copy() when you want an independent copy
a = {10, 20, 30}
b = a.copy()    # independent copy
b.add(40)
print(a)        # {10, 20, 30} — unchanged
print(b)        # {10, 20, 30, 40}

3. Set Operations

Sets support mathematical operations like union, intersection, difference, and symmetric difference.

Example 1: Union and Intersection

a = {1, 2, 3, 4}
b = {3, 4, 5, 6}

# Union — all elements from both sets
print(a | b)           # {1, 2, 3, 4, 5, 6}
print(a.union(b))      # same result

# Union of multiple sets
c = {6, 7, 8}
print(a | b | c)       # {1, 2, 3, 4, 5, 6, 7, 8}
print(a.union(b, c))   # same result

# Intersection — only elements in both sets
print(a & b)                  # {3, 4}
print(a.intersection(b))      # same result

# Intersection of multiple sets
x = {1, 2, 3}
y = {2, 3, 4}
z = {3, 4, 5}
print(x & y & z)              # {3}
print(x.intersection(y, z))   # same result

Example 2: Difference and Symmetric Difference

a = {1, 2, 3, 4}
b = {3, 4, 5, 6}

# Difference — elements in a but NOT in b
print(a - b)              # {1, 2}
print(a.difference(b))    # same result

# Order matters with difference
print(b - a)              # {5, 6}

# Symmetric difference — elements in either, but NOT both
print(a ^ b)                          # {1, 2, 5, 6}
print(a.symmetric_difference(b))      # same result

# Practical use case
students_math  = {"Alice", "Bob", "Carol", "Dave"}
students_science = {"Bob", "Carol", "Eve", "Frank"}

only_math    = students_math - students_science
only_science = students_science - students_math
both_classes = students_math & students_science

print(f"Only math: {only_math}")
print(f"Only science: {only_science}")
print(f"Both classes: {both_classes}")

Example 3: Subsets, Supersets, and Disjoint Sets

# issubset() — all elements of a are in b
a = {1, 2, 3}
b = {1, 2, 3, 4, 5}

print(a.issubset(b))    # True
print(a <= b)           # True (same as issubset)
print(b.issubset(a))    # False

# issuperset() — b contains all elements of a
print(b.issuperset(a))  # True
print(b >= a)           # True (same as issuperset)

# isdisjoint() — no elements in common
x = {1, 2, 3}
y = {4, 5, 6}
z = {3, 4, 5}

print(x.isdisjoint(y))  # True — no shared elements
print(x.isdisjoint(z))  # False — 3 is in both

# Practical example: permission checking
required_perms = {"read", "write"}
user_perms     = {"read", "write", "execute"}

print(required_perms.issubset(user_perms))  # True — user has access

4. Frozen Sets & Advanced Collections

Frozen sets are immutable sets. The collections module provides powerful specialized container types.

Example 1: Frozen Sets

# frozenset — immutable version of a set
numbers = frozenset([1, 2, 3, 4, 5])
print(numbers)  # frozenset({1, 2, 3, 4, 5})

# numbers.add(6)    # ❌ AttributeError — immutable
# numbers.remove(1) # ❌ AttributeError — immutable

# Frozen sets support all read operations
print(3 in numbers)       # True
print(len(numbers))       # 5

# Frozen sets can be used as dictionary keys (regular sets cannot)
permissions = {
    frozenset({"read"}):            "basic",
    frozenset({"read", "write"}):   "standard",
    frozenset({"read", "write", "admin"}): "full"
}
user_perms = frozenset({"read", "write"})
print(permissions[user_perms])  # standard

# Frozen set operations work just like regular sets
a = frozenset({1, 2, 3})
b = frozenset({2, 3, 4})
print(a | b)   # frozenset({1, 2, 3, 4})
print(a & b)   # frozenset({2, 3})

Example 2: Counter

from collections import Counter

# Count frequency of items in a list
data = ["apple", "banana", "apple", "orange", "banana", "apple"]
count = Counter(data)
print(count)
# Output: Counter({'apple': 3, 'banana': 2, 'orange': 1})

# Access count like a dictionary
print(count["apple"])    # 3
print(count["grape"])    # 0 — no KeyError for missing items

# Most common elements
print(count.most_common(2))
# Output: [('apple', 3), ('banana', 2)]

# Count characters in a string
letter_count = Counter("mississippi")
print(letter_count)
# Counter({'s': 4, 'i': 4, 'p': 2, 'm': 1})

# Arithmetic with Counters
a = Counter({"x": 3, "y": 2})
b = Counter({"x": 1, "z": 4})
print(a + b)   # Counter({'z': 4, 'x': 4, 'y': 2})
print(a - b)   # Counter({'y': 2, 'x': 2})

Example 3: defaultdict and namedtuple

from collections import defaultdict, namedtuple

# defaultdict — never raises KeyError for missing keys
# Provide a default type (int, list, set, etc.)
word_count = defaultdict(int)
words = ["cat", "dog", "cat", "bird", "dog", "cat"]
for word in words:
    word_count[word] += 1   # no KeyError even on first access
print(dict(word_count))
# Output: {'cat': 3, 'dog': 2, 'bird': 1}

# defaultdict with list — great for grouping
groups = defaultdict(list)
students = [("Math", "Alice"), ("Science", "Bob"), ("Math", "Carol")]
for subject, student in students:
    groups[subject].append(student)
print(dict(groups))
# Output: {'Math': ['Alice', 'Carol'], 'Science': ['Bob']}

# namedtuple — tuple with named fields, like a lightweight class
Student = namedtuple("Student", ["name", "age", "grade"])

s1 = Student("Alice", 20, "A")
s2 = Student(name="Bob", age=22, grade="B")

print(s1.name)    # Alice
print(s2.grade)   # B
print(s1[0])      # Alice — index access still works

# Practical: store and display records
students = [
    Student("Alice", 20, "A"),
    Student("Bob",   22, "B"),
    Student("Carol", 21, "A"),
]
for s in students:
    print(f"{s.name} (age {s.age}): Grade {s.grade}")
